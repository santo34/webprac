<script>

let html = document.getElementById("html");
let css = document.getElementById("css");
let js = document.getElementById("js");

let output = document.getElementById("output");

/* RUN CODE */

function runCode(){

let code = `

<style>
${css.value}
</style>

${html.value}

<script>
${js.value}
<\/script>

`;

output.srcdoc = code;

}

/* AUTO RUN */

html.addEventListener("input", runCode);
css.addEventListener("input", runCode);
js.addEventListener("input", runCode);

/* LINE NUMBERS */

function updateLines(textarea, lines){

let count = textarea.value.split("\n").length;

let numbers = "";

for(let i=1;i<=count;i++){
numbers += i + "<br>";
}

lines.innerHTML = numbers;

}

function setup(textareaId, lineId){

let textarea = document.getElementById(textareaId);
let lines = document.getElementById(lineId);

textarea.addEventListener("input", ()=>{

updateLines(textarea, lines);

});

updateLines(textarea, lines);

}

setup("html","html-lines");
setup("css","css-lines");
setup("js","js-lines");

/* OPEN PREVIEW */

function openPreview(){

let newWindow = window.open();

newWindow.document.write(output.srcdoc);

}

/* START */

runCode();

</script>
